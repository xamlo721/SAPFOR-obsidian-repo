Пример программы, которая хорошо подходит для тестирования распределения данных.

```fortran
PROGRAM JACOBY_SER
PARAMETER (L=4096, ITMAX=100)
REAL A(L,L), B(L,L)

PRINT *, '********** TEST_JACOBI ********'
DO IT = 1, ITMAX
	DO J = 2, L-1
		DO I = 2, L-1
			A(I,J) = B(I,J)
		ENDDO
	ENDDO

	DO J = 2, L-1
		DO I = 2, L-1
			B(I,J) = (A(I-1, J) + A(I, J-1) + A(I+1, J) + A(I, J+1)) / 4
		ENDDO
	ENDDO
ENDDO

PRINT *,B
END
```
